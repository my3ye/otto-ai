## Implementation Notes for `outreach_sender.py`

### Recommended logic

```python
def build_message(lead: dict) -> str:
    name = lead["business_name"] or lead["name"]
    city = lead.get("city") or "Sri Lanka"
    rating = lead.get("rating")
    reviews = lead.get("user_ratings_total") or 0
    lead_type = lead["lead_type"]

    if lead_type in ("no_website", "no-website"):
        if rating and rating >= 4.0 and reviews >= 10:
            return TEMPLATE_A1.format(
                business_name=name, city=city, rating=rating, reviews=reviews
            )
        else:
            return TEMPLATE_A2.format(business_name=name, city=city)

    elif lead_type == "revamp_candidate":
        # Check lead_notes to distinguish FB-only vs outdated site
        notes = lead.get("lead_notes", "") or ""
        if "facebook.com" in notes.lower() or "instagram.com" in notes.lower():
            return TEMPLATE_B1.format(business_name=name, city=city)
        else:
            return TEMPLATE_B2.format(business_name=name, city=city)

    else:
        return TEMPLATE_C1.format(business_name=name, city=city)
```
